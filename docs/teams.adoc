= Teams

== Create a team

[source,cypher]
----
:params {
  id: 'zZn839D0QIMG',
  name: 'Test Team',
  description: 'Employees from Team',
  public: true,
  open: true,
  admin: 'adam@neo4j.com',
  domains: ['neo4j.com']
}
----

[source,cypher]
----
CREATE (t:Team {
  id: $id, // a nanoid
  name: $name,
  description: $description,
  public: $public,
  open: $open,
  pin: left(toString(toInteger(rand()*10000000)), 6)
})
RETURN
  'https://graphacademy.neo4j.com/teams/'+ t.id AS link,
  t.pin AS pin
----

== Assign an admin

[source,cypher]
----
MATCH (u:User {email: $admin})
MATCH (t:Team {id $id})
MERGE (u)-[r:MEMBER_OF]->(t)
SET r.createdAt = datetime(),
  r.role = 'admin'
----
